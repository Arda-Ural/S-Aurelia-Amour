using System; using System.Collections.Generic; using System.Linq; class Program { // Kullanıcı Şablonu public class User { public int Id { get; set; } public string Username { get; set; } public string Password { get; set; } public string Role { get; set; } // "Admin" veya "User" } static List Veritabani = new List(); static User AktifKullanici = null; static void Main(string[] args) { // 1. ADIM: Başlangıçta bir admin ekleyelim ki sisteme girebilelim Veritabani.Add(new User { Id = 1, Username = "admin", Password = "123", Role = "Admin" }); while (true) { if (AktifKullanici == null) { // GİRİŞ YAPILMAMIŞSA Console.WriteLine("\n=== ANA MENÜ ==="); Console.WriteLine("1- Giriş Yap (admin/123 dene)"); Console.WriteLine("2- Kayıt Ol"); Console.Write("Seçim: "); string secim = Console.ReadLine(); if (secim == "1") Giris(); else if (secim == "2") Kayit(); } else if (AktifKullanici.Role == "Admin") { // ADMİN GİRİŞ YAPMIŞSA Console.WriteLine($"\n=== ADMİN PANELİ: {AktifKullanici.Username} ==="); Console.WriteLine("1- Tüm Kullanıcıları Listele"); Console.WriteLine("2- Birine Adminlik Yetkisi Ver"); Console.WriteLine("3- Kullanıcı Sil"); Console.WriteLine("4- Çıkış Yap"); Console.Write("Seçim: "); string secim = Console.ReadLine(); if (secim == "1") Listele(); else if (secim == "2") YetkiVer(); else if (secim == "3") Sil(); else if (secim == "4") AktifKullanici = null; } else { // NORMAL KULLANICI GİRİŞ YAPMIŞSA Console.WriteLine($"\n=== KULLANICI PANELİ: {AktifKullanici.Username} ==="); Console.WriteLine("1- Bilgilerimi Gör"); Console.WriteLine("2- Çıkış Yap"); Console.Write("Seçim: "); string secim = Console.ReadLine(); if (secim == "1") Console.WriteLine($"Ad: {AktifKullanici.Username}, Rol: {AktifKullanici.Role}"); else if (secim == "2") AktifKullanici = null; } } } static void Giris() { Console.Write("Kullanıcı Adı: "); string u = Console.ReadLine(); Console.Write("Şifre: "); string p = Console.ReadLine(); var user = Veritabani.FirstOrDefault(x => x.Username == u && x.Password == p); if (user != null) { AktifKullanici = user; Console.WriteLine("Giriş Başarılı!"); } else Console.WriteLine("Hatalı Giriş!"); } static void Kayit() { Console.Write("Yeni Ad: "); string u = Console.ReadLine(); Console.Write("Yeni Şifre: "); string p = Console.ReadLine(); int yeniId = Veritabani.Max(x => x.Id) + 1; Veritabani.Add(new User { Id = yeniId, Username = u, Password = p, Role = "User" }); Console.WriteLine("Kayıt Tamam."); } static void Listele() { Console.WriteLine("\n--- SİSTEMDEKİLER ---"); foreach (var u in Veritabani) Console.WriteLine($"ID: {u.Id} | Ad: {u.Username} | Rol: {u.Role}"); } static void YetkiVer() { Console.Write("Admin yapılacak ID: "); int id = int.Parse(Console.ReadLine()); var user = Veritabani.FirstOrDefault(x => x.Id == id); if (user != null) { user.Role = "Admin"; Console.WriteLine("Yetki Verildi!"); } } static void Sil() { Console.Write("Silinecek ID: "); int id = int.Parse(Console.ReadLine()); Veritabani.RemoveAll(x => x.Id == id); Console.WriteLine("Silindi."); } }
